# ES6 的基础知识

# 1. let、const

ES6 新增的用来创建变量和常量的

let a = 12;
a =13;
console.log(a); // 13

const b = 12;
b = 13; // Uncaught TypeError: Assignment to constant
variable,基于const创建变量，变量存储的值不能被修改（常量）

基于 ES6 中 let 来创建变量，是存储块级作用域（类似于私有作用域） 作用域（栈内存）

  1. 全局作用域
  2. 私有作用域（函数执行）
  3. 块级作用域（一般用大括号包起来的都是块级作用域，前提是 es6 语法糖）

全局作用域：js 运行在能解析的内核和引擎中，而我们现在接触到的就是浏览器，所以我们现在的 js 代码都在浏览器中运行的，浏览器给我们提供了一个供 js 代码执行的环境

let 和 var 的区别

  • let 不存在变量提升（当前作用域中，不能再 let 声明前使用变量）
  • 同一个作用中，let 不允许重复声明
  • let 解决了 typeof 的一个暂时性死区问题
  • 全局作用域中，使用 let 声明的变量并没有给 window 加上对应的属性
  • let 会存在块作用域（除对象以外的大括号都可以被看做块级私有作用域）

# 箭头函数及 this 问题

ES6 中新增了创建函数的方式，箭头函数

真实项目中是箭头函数和 function 这个普通函数混合使用

  • 箭头函数简化了创建函数的代码
// 箭头函数的创建都是函数表达式方式(变量=函数))，这种模式下，不存在变量提升，函数只能在创建完成后被执行（也就是创建的代码之后执行
const fn = ([形参]) => {
  // 函数体（return）
};

fn([实参]);

// 形参只有一个，小括号可以不加
const fn = n => {};

// 函数体重只有一句话，并且是return xxx的，可以省略大括号和return等可以省略
const fn = n => n * 10;

function fn(n) {
  return function(m) {
    return m + ++n;
  };
}

// 等价于上面的fn
const fn = n => m => m + ++n;

let fn = (context, ...arg) => {

# 解构赋值

让左侧出现和右侧值相同的结构，以此快速获取到我们需要的内容 真实项目中最常用的就是对数组和对象的解构赋值

拓展运算符：把剩下的内容存储到 x 中（x 是各数组），但是它只能出现在最后

let ary = [10, 20, 30, 40, 50];

// 1
let n = ary[0],
  m = ary[1],
  x = ary.slice(2);

// 2
let [n, m, ...x] = ary; // ...x拓展运算符
console.log(n, m, x);

// 3
let [n, , m, , , x = 0] = ary; //如果没有这一项，我们恶意基于等号赋值默认值

// 多维数组解构赋值
let ary = [10, [20, 30, [40, 50]]];
let [n, [, , [, m]]] = ary;
console.log(n, m); // 10 50

// 对象的解构赋值
let obj = {
  name: "xxx",
  age: 19,
  sex: "boy",
  friends: ["mike", "make", "cat"]
};
// 1
let name = obj.name;
let sex = obj.name;
// 2 创建的变量要和对象的属性名一致（默认）
let { name, sex } = obj;
console.log(name, sex); // name: 'xxx', sex: 'boy'
let {
  age: nianling // 冒号相当于给获取的结果设置了一个别名（变量名），创建了一个叫做nianling的变量存储于obj.age的值
} = obj;
console.log(nianling); // 19
let {
  height = "180CM" // 给获取的结果设置默认值（没有这个属性走的是默认值）
} = obj;
console.log(height); // 180cm
